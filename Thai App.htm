        let completedDays = [];

        function init() {

        // --- Chart Rendering ---

        let progressChartInstance;

        let vocabChartInstance;

            progressChartInstance = new Chart(ctxProgress, {

                        backgroundColor: ['#FACC15', '#CBD5E1'], // Yellow & Gray

                        borderWidth: 0

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: { legend: { position: 'bottom' } }

                }

            });

 

            // 2. Vocab Chart (Bar)

            const ctxVocab = document.getElementById('vocabChart').getContext('2d');

        function updateCharts() {

            const doneCount = completedDays.length;

            progressChartInstance.data.datasets[0].data = [doneCount, 30 - doneCount];

            progressChartInstance.update();

        }

 

        // --- Render Days ---

        // --- Interactions ---

        function toggleDetails(dayId) {

            const el = document.getElementById(`details-${dayId}`);

            el.classList.toggle('hidden');

        }

 

        function filterPhase(phase) {

            currentPhaseFilter = phase;

           

            // Visual feedback for buttons

            const buttons = document.querySelectorAll('#phase-filters button');

            buttons.forEach(btn => {

                btn.classList.remove('bg-blue-900', 'text-white', 'ring-2', 'ring-yellow-400');

                btn.classList.add('bg-white', 'text-blue-900', 'border-blue-200');

            });

            event.target.classList.remove('bg-white', 'text-blue-900', 'border-blue-200');

            event.target.classList.add('bg-blue-900', 'text-white', 'ring-2', 'ring-yellow-400');

 

            renderDays();

        }

 

        function checkAnswer(dayId, selectedIdx, correctIdx) {

            const feedbackEl = document.getElementById(`feedback-${dayId}`);

            feedbackEl.classList.remove('hidden');

            if (selectedIdx === correctIdx) {

                feedbackEl.innerHTML = "<span class='text-green-600'>မှန်ပါတယ်! (Correct) 🎉</span>";

            } else {

                feedbackEl.innerHTML = "<span class='text-red-500'>မှားနေပါတယ်၊ ပြန်ကြိုးစားကြည့်ပါ (Try again)</span>";

            }

        }

 

        function markComplete(dayId) {

            if (!completedDays.includes(dayId)) {

                completedDays.push(dayId);

                localStorage.setItem('thaiAppProgress', JSON.stringify(completedDays));

                updateCharts();

                updateProgressHeader();

                renderDays();

                // Keep expanded after render

                setTimeout(() => document.getElementById(`details-${dayId}`).classList.remove('hidden'), 50);

            }

        }

 

        function updateProgressHeader() {

            document.getElementById('header-progress').textContent = `${completedDays.length} / 30 Days`;

        }

 

        function loadProgress() {

            const saved = localStorage.getItem('thaiAppProgress');

            if (saved) {

                completedDays = JSON.parse(saved);

            }

        }

 

        window.onload = init;
